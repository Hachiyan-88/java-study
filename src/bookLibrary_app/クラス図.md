        + Book(int id, String title, String author)

        + getTitle() String
        + getAuthor() String

    class BookRepository {

        + findAll() List~Book~　　一覧

        + findByTitle(String title) List~Book~　タイトル検索
    }

    class LibraryApp {
        - Scanner scanner
        - BookRepository repository
        + start()
        + showMenu()
    }
